function trait(a,b){
   var sum =0
   for(let i=0;i<a.length;i++)
     { if(!(b.includes(a[i]))) 
        sum = sum+a[i]
     }
   for(let i=0;i<b.length;i++)
     { if(!(a.includes(b[i]))) 
        sum = sum+b[i]
     }
return sum      
 }
